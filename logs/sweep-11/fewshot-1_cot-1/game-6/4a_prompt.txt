([ > Sword | Stationary Obstacle ] -> [ Sword | Stationary Obstacle ])

[ > Player | Money ] -> [ | Player ] SFX5

late random [ Player Cave | ... | Cave ] -> [ Cave | ... | Player Cave ] SFX2

(late random [ Sword Cave | ... | Cave ] -> [ Cave | ... | Sword Cave ])
