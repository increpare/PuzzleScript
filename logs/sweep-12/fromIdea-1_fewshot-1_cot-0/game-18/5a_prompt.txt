(Wallify)

[ Moving Player ] [ H_step ] -> [ Moving Player ] [ Moving H_step ]

(gravity)
down [ Player no H_grav | no Solid ] -> [ Player H_grav | ]

[ up Player ] -> cancel

(movement/pushing)
Right [ > Player ] -> [ > PlayerR ]
Left [ > Player ] -> [ > PlayerL ]
