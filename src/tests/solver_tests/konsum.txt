======
RULES
======

[lines]->[]

[nosuck]->[]

[ > player | wall ] -> cancel

[action explore blacktile] -> [ place blacktile]
[stationary place | blacktile ] -> [ place | blacktile place]


(to place)

(1 must have at least one piece on black)
random  [ place blacktile] -> [ place blacktile canplace ]

[ canplace ] [ blacktile place | blacktile no place ] -> [ ] [ blacktile place | blacktile ]

[canplace] [ action place ] ->[canplace] [ action place blacktile ]

[canplace]->[]

random [ action place ] -> [ explore ]

[place ] [ explore ] -> [] [ explore ]

late [ explore no blacktile ] -> [ explore nosuck]


late up [ place | no place ] -> [ place y_up | ]
late down [ place | no place ] -> [ place y_down | ]
late left [ place | no place ] -> [ place y_left | ]
late right [ place | no place ] -> [ place y_right | ]

random late [ place blacktile ] -> [ place temp blacktile ]
late [temp]  [ place ] -> [ temp ] [ place temp ]

late [y_up no temp ] -> [ b_up ]
late [y_down no temp ] -> [ b_down ]
late [y_left no temp ] -> [ b_left ]
late [y_right no temp ] -> [ b_right ]


late [ temp] -> []

late up [ y_up blacktile | blacktile ] -> [ r_up blacktile | blacktile ]
late down [ y_down blacktile | blacktile ] -> [ r_down blacktile | blacktile ]
late left [ y_left blacktile | blacktile ] -> [ r_left blacktile | blacktile ]
late right [ y_right blacktile | blacktile ] -> [ r_right blacktile | blacktile ]
