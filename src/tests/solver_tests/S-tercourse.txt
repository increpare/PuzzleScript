[wall_s wall_w no wall_blot_ne ] -> [ wall_s wall_w wall_blot_ne ]

	down [no player_hat | action player ] -> [ action player_hat  | action player no player_start_location ]
    
    
	(Draw holding domain)
    right [ | action player | ] -> [ holding_domain | action player holding_domain | holding_domain ]
    down [ no holding_domain | holding_domain | no holding_domain ] -> [  | holding_domain | holding_domain ]
	
    right [action player] [ holding_domain | holding_domain no player | holding_domain ] -> [action player] [ attachment holding_domain | holding_domain | holding_domain attachment ]
    
    [holding_domain]->[]
)
