RULES
======

[ explode]->[no explode]
[ action Player | Crate ] -> [ action  Player | > Crate ]

up [ action Player | no wall ] -> [ action  Player |  explode_up ]
down [ action Player | no wall ] -> [ action  Player |  explode_down  ]
left [ action Player | no wall ] -> [ action  Player |  explode_left ]
right [ action Player | no wall ] -> [ action  Player |  explode_right ]

late [ explode Crate ] -> [crate]
