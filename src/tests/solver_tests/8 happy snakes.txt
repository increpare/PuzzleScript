LEGEND
=======

Barrier = HBarrier or VBarrier or HVBarrier

Player = Head

. = Background
# = Wall
P = Player

@ = Player and Target

Thing = Wall or Player or Body

worm = head or body

======
RULES
======

[badtarget]->[]
[goodtarget]->[]
[ > Head | Barrier ] -> cancel

[ > Head | No Thing ] -> [ Body | Head ]

[ > Head | Wall ] [ Body ] -> [ > Head | Wall ] [ < Body ]

[ > Body | Wall ] -> cancel

[ Wall | > Head | Wall ] -> cancel

[ | > Head | Wall ] -> [ T | Head | Wall ]

late [ T Body ] -> cancel
late [ T ] -> [ Body ]

late [ Collectable Worm ] -> [ Worm ] Sfx1

late [Starspawn] -> [Starspawn Collectable]
late [ worm no target ] [Starspawn Collectable] -> [ worm] [Starspawn]
late [ target no worm ] [Starspawn Collectable] -> [ target ] [Starspawn]

late [starspawn collectable ]->[collectable] sfx2

late [collectable] [target]->[collectable] []
late [collectable] [goodtarget]->[collectable] []
late [collectable] [badtarget]->[collectable] []

late [ worm target ] -> [ worm target goodtarget ]
late [ worm no target ] [ target] -> [worm] [target badtarget]
