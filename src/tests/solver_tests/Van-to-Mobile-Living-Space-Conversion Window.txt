RULES
======

[ >  Player | Crate ] -> [  >  Player | > Crate  ]
[ >  Crate | Crate ] -> [  >  Crate | > Crate  ]

late [ eye ] ->[]
late  [Player_G ]->[Player_G eye_l ]
late  [Player_b ]->[Player_b eye_r ]
late right [ eye_l | no eye_r ] -> [ | ]
late right [ no eye_l | eye_r ] -> [ | ]
late down [ eye | eye ] -> [ eye | ]

late down [player_g no eye_l | player_g no eye_l ] -> cancel

late down [ no player_g | Player_G | no Player_G ] -> cancel
late down [ no player_b | Player_b | no Player_b ] -> cancel
