RULES
======


up [ up player player_move_up | move_piece ] -> cancel
down [ down player player_move_down | move_piece ] -> cancel
left [ left player player_move_left | move_piece ] -> cancel
right [ right player player_move_right | move_piece ] -> cancel



[ up player no player_move_up ] -> cancel
[ down player no player_move_down ] -> cancel
[ left player no player_move_left ] -> cancel
[ right player no player_move_right ] -> cancel

[ > player | obstacle ] -> cancel
[ > player player_move ] -> [ > player > player_move ]

late up [ player_move_up | fire ] -> [ | fire ] sfx1
