======
RULES
======

[ action player ] -> cancel

[ > player | wall ] -> cancel
[ > player | crate | wall ] -> cancel
[ > player | crate | crate ] -> cancel


up 		[ up player 	lastup 		no laststart ] -> [ up player 		spawncrate ]
down 	[ down player 	lastdown 	no laststart ] -> [ down player 	spawncrate ]
left 	[ left player 	lastleft 	no laststart ] -> [ left player 	spawncrate ]
right 	[ right player 	lastright 	no laststart ] -> [ right player 	spawncrate ]

[ lastany ] -> [ ]

up 		[ up player 	| ] -> [ up player 		| lastup 	]
down 	[ down player 	| ] -> [ down player 	| lastdown 	]
left 	[ left player 	| ] -> [ left player 	| lastleft 	]
right 	[ right player 	| ] -> [ right player 	| lastright ]

[ >  Player | Crate ] -> [  >  Player | > Crate  ]

late [ spawncrate player ] -> cancel
late [ spawncrate ] -> [ crate ]

==============
WINCONDITIONS
==============

All Target on Crate
All Crate on Target
