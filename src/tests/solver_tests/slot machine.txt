======
RULES
======


[ htarget player_fall ] -> [  htarget ]

([ htarget player ] -> [ htarget player_fall ] sfx1)

[ htarget halfin_h ] -> [  htarget ]
[ vtarget halfin_v ] -> [  vtarget ]

[ coin_h htarget ] -> [ htarget halfin_h ] sfx0
[ coin_v vtarget ] -> [ vtarget halfin_v ] sfx0


horizontal [ > player  | coin_flat | no obstacle ] -> [ > player  | | coin_v ]
horizontal [ > player  | coin_v | no obstacle ] -> [ > player  | | coin_flat ]
horizontal [ > player  | coin_h | no obstacle ] -> [ > player  | | coin_h ]



vertical [ > player  | coin_flat | no obstacle ] -> [ > player  | | coin_h ]
vertical [ > player  | coin_h | no obstacle ] -> [ > player  | | coin_flat ]
vertical [ > player  | coin_v | no obstacle ] -> [ > player  | | coin_v ]

late [ shadow ] -> []

late down [ player | ] -> [ player | shadow_player ]
late down [ wall | ] -> [ wall | shadow_wall ] 

late down [ coin_flat | ] -> [ coin_flat | shadow_coin_flat ]
late down [ coin_h | ] -> [ coin_h | shadow_coin_h ]
late down [ coin_v | ] -> [ coin_v | shadow_coin_v ]
late [ coin_flat ] -> [ coin_flat coin_on_flatshadow ]
[]->again
