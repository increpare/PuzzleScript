=======
LEGEND
=======


Player = SnakeUp or SnakeDown or SnakeLeft or SnakeRight
Obstacle = SnakeBod or Wall
Color = red or blue
snaketotality = player or snakebod

PortalUp = PortalUpR or PortalUpB
PortalDown = PortalDownR or PortalDownB
PortalLeft = PortalLeftR or PortalLeftB
PortalRight = PortalRightR or PortalRightB

PortalR = PortalUpR or PortalDownR or PortalLeftR or PortalRightR
PortalB = PortalUpB or PortalDownB or PortalLeftB or PortalRightB

Portal = PortalUp or PortalDown or PortalLeft or PortalRight

======
RULES
======

[ player no color ] -> [player red]


[portal ] -> [old portal]


[portalleft]->[left portalleft]
[portalright]->[right portalright]
[portalup]->[up portalup]
[portaldown]->[down portaldown]

[player ] -> [ player temp]
up [ action snakeup temp | ... | obstacle ] -> [ action snakeup | ... | obstacle portaltarget ] 
down [ action snakedown temp | ... | obstacle ] -> [ action snakedown | ... | obstacle portaltarget ] 
left [ action snakeleft temp | ... | obstacle ] -> [ action snakeleft | ... | obstacle portaltarget ] 
right [ action snakeright temp | ... | obstacle ] -> [ action snakeright | ... | obstacle portaltarget ] 
[temp]->[]
[portaltarget portal ] -> cancel

[ > player | < portal ] -> [ > player | < portal enter ]

[ > portal no enter | no obstacle no player ] [enter] [player] [counter] -> [ > portal | exit ] [ enter ] [ snakebod ] [portalmade] 


[ up portal | exit ] -> [ up portal | snakeup ]
[ down portal | exit ] -> [ down portal | snakedown ]
[ left portal | exit ] -> [ left portal | snakeleft ]
[ right portal | exit ] -> [ right portal | snakeright ]
[color no player ] [ player ] -> [] [color player]

[ up player exit ] -> [ snakeup ]
[ down player exit ] -> [ snakedown ]
[ left player exit ] -> [ snakeleft ]
[ right player exit ] -> [ snakeright ]

[ > snakebod ] -> [snakebod]

[ > portal ] -> [ portal ]

[ > Player | Obstacle ] -> [ Player | Obstacle ]
up [ up Player Color | No Obstacle ] -> [ SnakeBod | SnakeUp Color ] 
down [ down Player Color | No Obstacle ] -> [ SnakeBod | SnakeDown Color ] 
left [ left Player Color | No Obstacle ] -> [ SnakeBod | SnakeLeft Color ] 
right [ right Player Color | No Obstacle ] -> [ SnakeBod | SnakeRight Color ] 

right [ action snakeright Red | ... | Obstacle No Portal ] -> [snakeright Blue | ... | Obstacle PortalLeftR ]
right [ action snakeright Blue | ... | Obstacle No Portal ] -> [snakeright Red | ... | Obstacle PortalLeftB ]

left [ action snakeleft Red | ... | Obstacle No Portal ] -> [snakeleft Blue | ... | Obstacle PortalRightR ]
left [ action snakeleft Blue | ... | Obstacle No Portal ] -> [snakeleft Red | ... | Obstacle PortalRightB ]

up [ action snakeup Red | ... | Obstacle No Portal ] -> [snakeup Blue | ... | Obstacle PortalDownR ]
up [ action snakeup Blue | ... | Obstacle No Portal ] -> [snakeup Red | ... | Obstacle PortalDownB ]

down [ action snakedown Red | ... | Obstacle No Portal ] -> [snakedown Blue | ... | Obstacle PortalupR ]
down [ action snakedown Blue | ... | Obstacle No Portal ] -> [snakedown Red | ... | Obstacle PortalupB ]



[old portalB] [no old portalB] -> [] [portalB]
[old portalR] [no old portalR] -> [] [portalR]
[old] -> []

[enter] -> []
[exit] -> []

[portaltarget]->[]
