(pressing action when you're right above a foot is the same as pressing up)
down [ action player | foot_inactive ] -> [ up player | foot_inactive ]



(2 -  raise unobstructed active leg, inactive leg grounded)
down [ up player no active_foot_raise_obstructed inactive_grounded| active | active ] [ foot_active ] -> [ up player | | active ] [ up foot_active ]

(note to self: moved the following line up here from somewhere near the end. hope it doesn't break things, but if something breaks its probably this)
(propagate foot movements to leg)
[up foot_active ] [ leg_active ] -> [ up foot_active ] [ up leg_active ]

(3 - lower ungrounded active leg into space)
random down [ down player no active_grounded ] [ foot_active ]  [ leg_active | no leg_active ]  -> [ down player no active_grounded ][ down foot_active ]  [ leg_active | leg_active ] 




(4 - raise active leg (inactive leg ungrounded) = lower inactive leg + lower head )
	(tell every non-active thing to move down)
down [ up player no inactive_grounded | active ] [ dog_not_active no player ] ->  [ up player no inactive_grounded | active ] [ down dog_not_active ]
	(remove topmost active leg segment)
down [ up player no inactive_grounded | active ] -> [ | player ]

(5 - lower grounded active leg pushing the body up)
down [ down player active_grounded ] [ dog_not_active no player ] ->  [ down player active_grounded ]  [ up dog_not_active ]
down [ | face_tr | down player active_grounded ] -> [ face_tr |  player active_grounded | leg_active ]


(6 - walk forward (active grounded, inactive ungrounded))
	(inactive foot has to be not just ungrounded, but above the active one (to ensure it can move without friction))
    down [ right player active_grounded  ] [  foot_inactive | no leg_active ] -> cancel
    
    
	(first all faces try to move right)
	(remove movement if activity is wrong)
right [ right player no active_grounded ] -> cancel
( right [ right player inactive_grounded ] -> cancel )

	(move face)
    
[ right player active_grounded  ] [ face ] -> [ right player active_grounded  no inactive_grounded ] [ right face ] 
right [ right face | no face ] -> [ | face moved_inactive ]

right [player moved_inactive ] [ leg_inactive | | ] -> [player moved_inactive ]  [ | | leg_active ]
right [player moved_inactive ] [ foot_inactive | | ] -> [player moved_inactive ]  [ | | foot_active ]

(turn active leg inactive (active->inactive))
down [ face_bl moved_inactive | leg_active ] -> [ face_bl moved_inactive | leg_inactive ]
down [ leg_inactive | leg_active ] -> [ leg_inactive | leg_inactive ]
right [leg_inactive | foot_active ] -> [ leg_inactive | foot_inactive ]

(remove moved tags)
[moved_inactive]->[]

(7 - walk back (active ungrounded, inactive grounded))
	(inactive foot has to  be not just ungrounded, but above the active one (to ensure it can move without friction))
    down [ left player inactive_grounded  ] [ foot_inactive leg_active ] -> cancel
    
	(first all faces try to move left)
	(remove movement if activity is wrong)
	(right [ right player no active_grounded ] -> cancel)
	(right [ right player inactive_grounded ] -> cancel)


(toggle frontlegforward)
[left player no frontlegforward inactive_grounded ] -> [ left player left frontlegforward inactive_grounded]
[left player stationary frontlegforward inactive_grounded ] -> [ left player inactive_grounded ]

	(move face)
[ left player inactive_grounded ] [ face ] -> [  left player inactive_grounded no active_grounded] [ left face ] 
left [ left face | no face ] -> [ | face moved_inactive ]


	(move front leg back (active->inactive) )
right [player moved_inactive ] [ | | leg_active ] -> [player moved_inactive ]  [ leg_inactive | | ]
right [player moved_inactive ] [ | | foot_active ] -> [player moved_inactive ]  [ foot_inactive | | ]


	(turn inactive leg active (inactive->active))

+ down [ leg_active | no leg_active | leg_inactive ] -> [ leg_active | leg_active | leg_active ]
right [leg_active | foot_inactive ] -> [ leg_active | foot_active ]
