======
RULES
======

up [  crate p_u tick | no inanimate ] ->  [ p_u > crate | ]
down [  crate p_d tick | no inanimate ] -> [ p_d > crate | ]
left [  crate p_l tick | no inanimate ] -> [ p_l > crate | ]
right [  crate p_r tick | no inanimate ] -> [ p_r > crate | ]

[> crate | player ] -> [ > crate | > player ]
[tick]->[]


[ > player | crate ] -> [ > player | > crate > tick] again

[ > obstacle | stationary obstacle ] -> cancel


late up [  crate p_u tick |  inanimate ] ->  [ p_u  crate |inanimate ]
late down [  crate p_d tick | inanimate ] -> [ p_d  crate | inanimate]
late left [  crate p_l tick | inanimate ] -> [ p_l  crate | inanimate]
late right [  crate p_r tick | inanimate ] -> [ p_r  crate | inanimate]
