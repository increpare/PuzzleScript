======
RULES
======

[ > player | crate ] -> [ > player | > crate ]
[ > crate | crate ] -> [ > crate | > crate ]


[ > crate | stationary obstacle ] -> [ crate | stationary obstacle ]
[ > player | stationary obstacle ] -> cancel

([ > player | obstacle | obstacle ] -> cancel)
[ > player | wall ] -> cancel 

late [ anytele ] -> []
late up [ crate | no obstacle ] -> [ crate | teleport_down ]
late down [ crate | no obstacle ] -> [ crate | teleport_up ]
late left [ crate | no obstacle ] -> [ crate | teleport_right ]
late right [ crate | no obstacle ] -> [ crate | teleport_left ]

late [ teleport_up teleport_down ] -> [ teleport_bad ]
late [ teleport_up teleport_left ] -> [ teleport_bad ]
late [ teleport_up teleport_right ] -> [ teleport_bad ]
late [ teleport_down teleport_left ] -> [ teleport_bad ]
late [ teleport_down teleport_right ] -> [ teleport_bad ]
late [ teleport_left teleport_right ] -> [ teleport_bad ]

late [ teleport_bad teleport_up ] -> [ teleport_bad ]
late [ teleport_bad teleport_down ] -> [ teleport_bad ]
late [ teleport_bad teleport_left ] -> [ teleport_bad ]
late [ teleport_bad teleport_right ] -> [ teleport_bad ]

late up [ teleport_up | crate | obstacle ] -> [teleport_bad|Crate|obstacle]
late down [ teleport_down | crate | obstacle ] -> [teleport_bad|Crate|obstacle]
late left [ teleport_left | crate | obstacle ] -> [teleport_bad|Crate|obstacle]
late right [ teleport_right | crate | obstacle ] -> [teleport_bad|Crate|obstacle]

random late up [ teleport_up player | crate | no obstacle ] -> [ teleport_up | crate | Player ] sfx1
+ late down [ teleport_down player | crate | no obstacle ] -> [ teleport_down | crate | Player ] sfx1
+ late left [ teleport_left player | crate | no obstacle ] -> [ teleport_left | crate | Player ] sfx1
+ late right [ teleport_right player | crate | no obstacle ] -> [ teleport_right | crate | Player ] sfx1
