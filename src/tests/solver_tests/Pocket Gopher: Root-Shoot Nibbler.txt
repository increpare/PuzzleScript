RULES
======

(don't forget about the case where you're chasing your tail)
( [ > player | vine ] -> cancel )

[ moving player ] -> again

[vine_n]->[vine_n n]
[vine_s]->[vine_s s]
[vine_e]->[vine_e e]
[vine_w]->[vine_w w]

[vine_ns]->[vine_ns n s]
[vine_ne]->[vine_ne n e]
[vine_nw]->[vine_nw n w]

[vine_se]->[vine_se s e]
[vine_sw]->[vine_sw s w]

[vine_ew]->[vine_ew e w]

[vine_nse]->[vine_nse n s e]
[vine_nsw]->[vine_nsw n s w]
[vine_new]->[vine_new n e w]
[vine_sew]->[vine_sew s e w]

[vine_nsew]->[vine_nsew n s e w]

[ > player | vine_end ] -> [ > player | no n no s no e no w ] sfx0 (munch)

[ > player | vine_body ] -> [ > player | > vine_body ] sfx1

[ > vine | vine ] -> [ > vine | > vine ]
+ up [ n moving vine | vine ] -> [ n moving vine | moving vine ]
+ down [ s moving vine | vine ] -> [ s moving vine | moving vine ]
+ left [ w moving vine | vine ] -> [ w moving vine | moving vine ]
+ right [ e moving vine | vine ] -> [ e moving vine | moving vine ]

[ > vine | wall ] -> cancel
[ > player | wall ] -> cancel
[ > player | nothingness ] -> cancel

up [ n | no s ] -> [  spawn | ]
down [ s | no n ] -> [  spawn | ]
left [ w  | no e ] -> [  spawn | ]
right [ e  | no w ] -> [  spawn | ]

[ n s e spawn ] -> [ vine_nse ]
[ n s w spawn ] -> [ vine_nsw ]
[ n e w spawn ] -> [ vine_new ]
[ s e w spawn ] -> [ vine_sew ]

[ n s spawn ] -> [ vine_ns ]
[ n e spawn ] -> [ vine_ne ]
[ n w spawn ] -> [ vine_nw ]

[ s e spawn ] -> [ vine_se ]
[ s w spawn ] -> [ vine_sw ]

[ e w spawn ] -> [ vine_ew ]

[ n spawn ] -> [ vine_n ]
[ s spawn ] -> [ vine_s ]
[ e spawn ] -> [ vine_e ]
[ w spawn ] -> [ vine_w ]

[  spawn ] -> [ vine_empty ]

[dir]->[]

late [vine_n]->[vine_n n]
late [vine_s]->[vine_s s]
late [vine_e]->[vine_e e]
late [vine_w]->[vine_w w]

late [vine_ns]->[vine_ns n s]
late [vine_ne]->[vine_ne n e]

late [vine_new]->[vine_new n e w]
late [vine_sew]->[vine_sew s e w]

late [vine_nsew]->[vine_nsew n s e w]

late up [ n vine no fallen | vine ] -> [ n vine | no fallen vine ]
+ late down [ s  vine no fallen | vine ] -> [ s vine | no fallen vine ]

+ late right [ e  vine no fallen | vine ] -> [ e vine | no fallen vine ]
