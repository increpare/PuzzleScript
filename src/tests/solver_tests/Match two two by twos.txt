[todestroy]->[]

(need to repeat this for down direction)
right [ red 	| | red ] 		-> [ ml red 	| | mr red ]
right [ green 	| | green ] 	-> [ ml green 	| | mr green ]
right [ blue	| | blue ] 		-> [ ml blue  	| | mr blue ]
right [ yellow 	| | yellow ] 	-> [ ml yellow 	| | mr yellow ]

right [ ml | ml | mr | mr ] -> [ ml mta | ml mtb | mr mta | mr mtb ]

[ml]->[]
[mr]->[]


down [ red 	| | red ] 		-> [ ml red 	| | mr red ]
down [ green 	| | green ] 	-> [ ml green 	| | mr green ]
down [ blue	| | blue ] 		-> [ ml blue  	| | mr blue ]
down [ yellow 	| | yellow ] 	-> [ ml yellow 	| | mr yellow ]

down [ ml | ml | mr | mr ] -> [ ml mt2a | ml mt2b | mr mt2a | mr mt2b ]


down [ mta | mta ] -> [ mta no color todestroy | mta no color todestroy] sfx1 again
down [ mtb | mtb ] -> [ mtb no color todestroy | mtb no color todestroy ] sfx1 again
right [ mt2a | mt2a ] -> [ mt2a no color todestroy | mt2a no color todestroy ] sfx1 again
right [ mt2b | mt2b ] -> [ mt2b no color todestroy | mt2b no color todestroy ] sfx1 again

[anymatch]->[]


right [ midgobble | ... | color | no color ] -> [ midgobble | ... | | color ]
right [ midgobble color | no color ] -> [ midgobble | color ]

up [ gobble | ... | color | no color ] -> [ gobble | ... | | color ]
up [ gobble color | no color ] -> [ gobble | color ]



[action player color ] [gobble ] -> [ player moved] [ gobble color] again


down [ color | gobble no color ] -> [  | gobble color ]
down [ color | no color | ... | gobble ] -> [ | color | ... | gobble ]


left [ color | midgobble no color ] -> [  | midgobble color ]
left [ color | no color | ... | midgobble ] -> [ | color | ... | midgobble ]

[action player no color ] [gobble color] -> [ action player color] [ gobble] sfx0 again

down [ color | gobble no color ] -> [  | gobble color ]
down [ color | no color | ... | gobble ] -> [ | color | ... | gobble ]


left [ color | midgobble no color ] -> [  | midgobble color ]
left [ color | no color | ... | midgobble ] -> [ | color | ... | midgobble ]
