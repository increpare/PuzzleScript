RULES

[pixel]->[]

right [stationary player | player ] -> [ player con_r | player con_l ]
down [stationary player | player ] -> [ player con_d | player con_u ]


right [knife_r | ] -> [ knife_r | knife_l]
down [knife_b | ] -> [ knife_b | knife_t]


[ > player | wall ] -> [ player | wall ]
+ [ right player knife_r ] -> [ stationary player knife_r]
+ [ left player knife_l ] -> [ stationary player knife_l]
+ [ up player knife_t ] -> [ stationary player knife_t]
+ [ down player knife_b ] -> [ stationary player knife_b]

+ up [ stationary player con_u | player ] -> [ stationary player con_u | stationary player ]
+ down [ stationary player con_d | player ] -> [ stationary player con_d | stationary player ]
+ left [ stationary player con_l | player ] -> [ stationary player con_l | stationary player ]
+ right [ stationary player con_r | player ] -> [ stationary player con_r | stationary player ]

+ [ > player | stationary player ] -> [ player | player ]


[ > player connection ] -> [ > player > connection ]



[ stationary player connection ] -> [ player stationary connection ]


late [ knife_r con_r ] -> [ knife_r cut_r ] sfx1
late [ knife_l con_l ] -> [ knife_l cut_l ] sfx1

late [ knife_t con_u ] -> [ knife_t cut_t ] sfx1
late [ knife_b con_d ] -> [ knife_b cut_b ] sfx1


late down [ con_r | con_u cut_r ] -> [ cut_r | con_u cut_r ]
late down [ con_l | con_u cut_l ] -> [ cut_l | con_u cut_l ]

late up [ con_r | con_d cut_r ] -> [ cut_r | con_d cut_r ]
late up [ con_l | con_d cut_l ] -> [ cut_l | con_d cut_l ]

late right [ con_u | con_l cut_t ] -> [ cut_t | con_l cut_t ]
late right [ con_d | con_l cut_b ] -> [ cut_b | con_l cut_b ]

late left [ con_u | con_r cut_t ] -> [ cut_t | con_r cut_t ]
late left [ con_d | con_r cut_b ] -> [ cut_b | con_r cut_b ]

late [ cut ]->[]

late [ player no hole ] -> [ player supported ]

late up [ supported con_u | ] -> [ supported con_u | supported ]
+ late down [ supported con_d | ] -> [ supported con_d | supported ]
+ late left [ supported con_l | ] -> [ supported con_l | supported ]
+ late right [ supported con_r | ] -> [ supported con_r | supported ]

late [ player no supported ] -> [ no connection ] sfx0
