[tl]->[tl exploding]

down [ wall no bond_d | fallblocker ] -> [ wall nofall | fallblocker ]

+ down [nofall bond_d | ] -> [ nofall bond_d | nofall ]
+ left [nofall bond_l | ] -> [ nofall bond_l | nofall ]
+ right [nofall bond_r | ] -> [ nofall bond_r | nofall ]


down [ wall no nofall | no fallblocker ] -> [ wall fall | ]

up [fall bond_u | ] -> [ fall bond_u | fall ]
+ down [fall bond_d | ] -> [ fall bond_d | fall ]
+ left [fall bond_l | ] -> [ fall bond_l | fall ]
+ right [fall bond_r | ] -> [ fall bond_r | fall ]

[fall wall]->[down wall] sfx1  again

[fall]->[]
[nofall]->[]

[up wall bond ] -> [ up wall up bond ]
+ [down wall bond ] -> [ down wall down bond ]
+ [left wall bond ] -> [ left wall left bond ]
+ [right wall bond ] -> [ right wall right bond ]

[target]->[]


[horizontal  player ] -> cancel

[up player_h ] -> [ up player_v ]
[down player_h ] -> [ down player_v ]

[right player_v ] -> [ right player_h ]
)

[player_h ] -> [ player_h htarget ]
[player_v ] -> [ player_v vtarget ]

horizontal [ htarget | no htarget no fenster ] -> [ htarget | htarget ]
vertical [ vtarget | no vtarget no fenster ] -> [ vtarget | vtarget ]


up [ bond_u | explosion ] -> [ | explosion ]
down [ bond_d | explosion ] -> [ | explosion ]
left [ bond_l | explosion ] -> [ | explosion ]
right [ bond_r | explosion ] -> [ | explosion ]

[explosion wall ] -> [ no bond ] sfx1 again

[explosion ] -> [ ] again
